% ##### custom commands #####
\newcommand{\pictureat}[4]{%
	\only<+(1)->{
		\begin{textblock*}{\paperwidth}(#1, #2)%
			\includegraphics[#3]{#4}%
		\end{textblock*}}}
		
% This custom command can be used with the theme and provides a zoom in of the specified picture
% it has to be specified manually by the first command if width = \pagewidth or height = \pageheight
% to fit on the screen. The second command specifies the path to the picture.
\newcommand{\fillpic}[2]{
	\clearpage
	\begin{textblock*}{\paperwidth}(0pt,0pt)
		\begin{tikzpicture}
		\useasboundingbox (0, 0) rectangle(\paperwidth,\paperheight);
		\fill [color = black, fill opacity=0.75](0, 0) rectangle(\paperwidth,\paperheight);
		\node (label) at (0.5\paperwidth,0.5\paperheight){\includegraphics[#1]{#2}};
		\end{tikzpicture}
	\end{textblock*}
}

	\begin{frame}{Background}{\& personal inspiration}

		It's advantages are:
		\begin{itemize}
			\item clean structure
			\item nice \& consistent design
			\item easy to use with a small basic \LaTeX{} knowledge
			\item useful additional commands
			\item export file is \texttt{.pdf} which can be opened nearly anywhere!
		\end{itemize}
	\end{frame}

		\vfill

		One of the most common ways to create slides is to use lots of itemizes.
		
		They are used for:
		\begin{itemize}

					\item topics

				\end{itemize}
			\end{itemize}
			\item <2-> they also can be made visible item by item
			\item <3-> like this!
		\end{itemize}
